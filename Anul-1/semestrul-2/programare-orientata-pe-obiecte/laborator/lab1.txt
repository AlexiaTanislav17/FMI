class SirInt{

};

class MatriceInt{
private:
    int n;
    SirInt *p;
public:
    MatriceInt(){
        n=0;
        p= nullptr;
    }
    
    ~MatriceInt(){
        // optionala
        this->n=0;
        delete []p;
    }
    
    friend std::istream& operator>>(std::istream &in, MatriceInt &aux){
        std::cout<<"n=";
        in>>aux.n
        aux.p = new SirInt(aux.n);
        for (int i=0; i<aux.n; i++){
            in >> aux.p;
        }
        
        
    }

};
