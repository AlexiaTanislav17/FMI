#include <iostream>
#include <vector>
#include <climits>
using namespace std;

void floydWarshall(int n, vector<vector<int>> &graph) {

    vector<vector<int>> dist = graph;

 
    for (int k = 0; k < n; k++) { 
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) { 
                if (dist[i][k] != INT_MAX && dist[k][j] != INT_MAX) {
                    dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
                }
            }
        }
    }


    cout << "Shortest distances between all pairs of nodes:\n";
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (dist[i][j] == INT_MAX) {
                cout << "INF ";
            } else {
                cout << dist[i][j] << " ";
            }
        }
        cout << endl;
    }
}

int main() {
    int n = 4;
    vector<vector<int>> graph = {
        {0, 3, INT_MAX, 5},
        {2, 0, INT_MAX, 4},
        {INT_MAX, 1, 0, INT_MAX},
        {INT_MAX, INT_MAX, 2, 0}
    };

    floydWarshall(n, graph);

    return 0;
}
